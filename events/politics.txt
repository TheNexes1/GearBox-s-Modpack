# Civil War for monarchy
country_event = {
	id = political.53
	title = political.53.t
	desc = political.53.d
	picture = GFX_report_event_fascist_gathering
	
	trigger = {
		has_idea_with_trait = fascist_demagogue
		NOT = { has_government = monarchy }
		monarchy > 0.25
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.75
			has_country_flag = monarchy_popular_movement
		}
	}
	
	option = {
		name = political.53.a
		if = {
			limit = {
				has_government = democratic
			}
			if = {
				limit = { monarchy > 0.4 }
				start_civil_war = {
					ruling_party = monarchy
					ideology = democratic
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchy > 0.4 } }
				start_civil_war = {
					ruling_party = monarchy
					ideology = democratic
					size = 0.7
				}
			}
		}
		limit = {
				has_government = fascism
			}
			if = {
				limit = { monarchy > 0.4 }
				start_civil_war = {
					ruling_party = monarchy
					ideology = fascism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchy > 0.4 } }
				start_civil_war = {
					ruling_party = monarchy
					ideology = fascism
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			if = {
				limit = { monarchy > 0.4 }
				start_civil_war = {
					ruling_party = monarchy
					ideology = communism
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchy > 0.4 } }

					ideology = communism
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = neutrality
			}

				limit = { monarchy > 0.4 }

					ruling_party = monarchy

					size = 0.6
				}
			}
			if = {
				limit = { NOT = { monarchy > 0.4 } }
				start_civil_war = {
					ruling_party = monarchy
					ideology = neutrality
					size = 0.7
				}
			}
		}
		remove_ideas = monarchy_revolutionaries
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
	option = {
		name = political.53.b
	}
